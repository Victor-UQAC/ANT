
global

    chroot /var/lib/haproxy

    stats timeout 30s
    user haproxy
    group haproxy
    daemon

    log global
    mode tcp
    option tcplog

    mode tcp
    default_backend mysql_servers

    mode tcp
    balance roundrobin
    server mysql1 maria1.ant.local:27017 check
